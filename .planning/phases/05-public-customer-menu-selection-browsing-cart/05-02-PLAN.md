    Create `src/app/r/[restaurantSlug]/page.tsx` as an async Server Component (NO `export const dynamic` — Next.js 16 is dynamic by default; adding force-dynamic is a forbidden no-op per RESEARCH).

    ```tsx
    import { notFound, redirect } from 'next/navigation'
    import { getRestaurantBySlug, getUnitsForRestaurant } from '@/lib/menu/queries'
    import UnitPicker from './unit-picker'

    export default async function RestaurantPage({
      params,
    }: { params: Promise<{ restaurantSlug: string }> }) {
      const { restaurantSlug } = await params
      const restaurant = await getRestaurantBySlug(restaurantSlug)
      if (!restaurant) notFound() // D-12: nonexistent or inactive

      const restaurantUnits = await getUnitsForRestaurant(restaurant.id)

      if (restaurantUnits.length === 0) {
        // MENU-06: valid restaurant, no units yet — graceful empty state, NOT a 404.
        return (
          <main className="flex min-h-screen flex-col items-center justify-center gap-2 p-8 text-center">
            <h1 className="text-xl font-semibold">{restaurant.name}</h1>
            <p className="text-muted-foreground">Este restaurante ainda não tem unidades cadastradas.</p>
          </main>
        )
      }

      if (restaurantUnits.length === 1) {
        redirect(`/r/${restaurantSlug}/${restaurantUnits[0].slug}`) // D-02
      }

      return <UnitPicker restaurantSlug={restaurantSlug} restaurantName={restaurant.name} units={restaurantUnits} /> // D-01/D-03/D-04
    }
    ```
    Pass plain serializable props to UnitPicker (units array, restaurantSlug, restaurantName) — do not pass the raw Drizzle objects with non-serializable fields; the unit rows are already plain objects so spreading is fine. CRITICAL: 0 units => empty state (NOT notFound), 1 unit => redirect, 2+ => picker. Do not collapse the 0-units branch into notFound.

    Props: `{ restaurantSlug: string; restaurantName: string; units: Array<{ id: string; name: string; slug: string; address: string | null; hours: string | null; lat: number | null; lng: number | null }> }`.

    Behavior:
    1. State `sortedUnits` initialized to `units` (unsorted fallback = D-01 default).
    2. On mount (`useEffect`), request geolocation:
       ```tsx
       if (!navigator.geolocation) return // silent fallback
       navigator.geolocation.getCurrentPosition(
         (pos) => {
           const me = { lat: pos.coords.latitude, lng: pos.coords.longitude }
           const withDist = [...units].sort((a, b) => {
             const da = a.lat != null && a.lng != null ? haversineDistanceKm(me, { lat: a.lat, lng: a.lng }) : Infinity
             const db = b.lat != null && b.lng != null ? haversineDistanceKm(me, { lat: b.lat, lng: b.lng }) : Infinity
             return da - db
           })
           setSortedUnits(withDist)
         },
         () => { /* PERMISSION_DENIED(1)/POSITION_UNAVAILABLE(2)/TIMEOUT(3): keep unsorted, no error UI */ },
         { timeout: 8000, maximumAge: 60_000 },
       )
       ```
       Import `haversineDistanceKm` from `@/lib/menu/format`. Units with null lat/lng sort to the END (Infinity distance) — never do null arithmetic. Any error path keeps the original unsorted list with NO error banner (D-03 graceful fallback).

Use an onClick handler on the Link, or a button that calls the write then router.push — either is acceptable as long as the localStorage write uses the exact key prefix `boamidia:lastUnit:` + restaurantSlug.
